static NXObject selectFace(out Selection.Response resp, out Point3d cursor)
        {
            NXObject selobj;
            Selection.SelectionType[] typeArray = { Selection.SelectionType.Faces };

            resp = _UI.SelectionManager.SelectObject("¬ыберите место на нижней плоскости паза", "¬ыберите место на нижней плоскости паза",
                Selection.SelectionScope.AnyInAssembly, false, typeArray, out selobj, out cursor);

            {
                return selobj;
            }
            else

            }
        }
